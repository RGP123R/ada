#include <stdio.h>
#include <stdlib.h>
#include <time.h> // Include time.h for clock()

#define MAX 5000

void selectionsort(int a[], int n) {
    int i, j, min;
    for (i = 0; i < n-1; i++) { // Corrected loop condition
        min = i;
        for (j = i+1; j < n; j++) { // Corrected loop condition
            if (a[j] < a[min])
                min = j;
        }
        // Swap the found minimum element with the first element
        int temp = a[min];
        a[min] = a[i];
        a[i] = temp;
    }
}

int main() {
    int a[MAX], n, i;

    printf("Enter number of elements to be sorted: ");
    scanf("%d", &n);

    if (n > MAX) {
        printf("Size exceeds the maximum limit.\n");
        return 1;
    }

    srand(time(NULL)); // Seed the random number generator

    for (i = 0; i < n; i++)
        a[i] = rand() % n;

    printf("UNSORTED ARRAY\n");
    for (i = 0; i < n; i++)
        printf("%d\t", a[i]);

    // Measure time before sorting
    clock_t start_time = clock();
    selectionsort(a, n);
    // Measure time after sorting
    clock_t end_time = clock();

    // Calculate time taken
    double time_taken = ((double) (end_time - start_time)) / CLOCKS_PER_SEC;
    printf("\nTime taken to sort: %.6f seconds\n", time_taken);

    printf("\nSorted array:\n");
    for (i = 0; i < n; i++)
        printf("%d\t", a[i]);
    
    printf("\n"); // Add newline for better formatting

    return 0;
}
