#include <stdio.h>

#define MAX 10

int s[MAX], x[MAX], d;
// Function to find subsets that sum to d
void sumofsub(int p, int k, int r) {
    int i;
    x[k] = 1; // Include the k-th element in the subset
    // Check if the current subset's sum is equal to d
    if ((p + s[k]) == d) {
        for (i = 1; i <= k; i++) {
            if (x[i] == 1) {
                printf("%d ", s[i]);
            }
        }
        printf("\n");
    } else if (p + s[k] + s[k + 1] <= d) {
        // If including the next element does not exceed d
        sumofsub(p + s[k], k + 1, r - s[k]);
    }
    // Check if excluding the k-th element is necessary and feasible
    if ((p + r - s[k] >= d) && (p + s[k + 1] <= d)) {
        x[k] = 0; // Exclude the k-th element from the subset
        sumofsub(p, k + 1, r - s[k]);
    }
}
int main() {
    int i, n, sum = 0;
    printf("\nEnter the number of elements: ");
    scanf("%d", &n);
    printf("\nEnter the set in increasing order: ");
    for (i = 1; i <= n; i++) {
        scanf("%d", &s[i]);
    }
    printf("\nEnter the maximum subset value: ");
    scanf("%d", &d);
    for (i = 1; i <= n; i++) {
        sum += s[i];
    }
    if (sum < d || s[1] > d) {
        printf("\nNo subset possible\n");
    } else {
        sumofsub(0, 1, sum);
    }
    return 0;
}
